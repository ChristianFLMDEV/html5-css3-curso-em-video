class AprendendoHtml:
    def __init__(self, nome, aula):
        self.nome = nome
        self.aula = aula

    def __str__(self):
        if aula >= 12:
            return(f"Boa {self.nome}, está avançando bem no curso de HTML5")
        else:
            return(f"Em breve você estará mais longe!")

aluno = AprendendoHtml("Christian", 8)
print(aluno)
